        function renderQuestions(tabId, startNum, endNum) {

            const container = document.getElementById('questions-' + tabId);

            const relevantQuestions = questions.filter(q => q.num >= startNum && q.num <= endNum);

            

            container.innerHTML = relevantQuestions.map((q) => `

                <div class="question-card">

                    <span class="question-number">問${q.num}</span>

                    <div class="question-text">${q.text}</div>

                    <div class="options">

                        ${q.options.map((opt, i) => `

                            <div class="option">

                                <input type="radio" id="q${q.num}_${i}" name="q${q.num}" value="${i + 1}" onchange="checkAnswer(${q.num}, ${q.answer})">

                                <label for="q${q.num}_${i}"><span>${i + 1}. ${opt}</span></label>

                            </div>

                        `).join('')}

                    </div>

                    <div class="result-message" id="result-${q.num}"></div>

                    <div class="answer-section" id="answer-${q.num}">

                        <span class="answer-label">正答</span>

                        <div class="answer-text"><strong>第${q.answer}選択肢：${q.options[q.answer - 1]}</strong></div>

                        <span class="answer-label" style="background: var(--color-warning); margin-top: 12px;">解説</span>

                        <div class="explanation">${q.explanation}</div>

                    </div>

                </div>

            `).join('');

        }


        function switchTab(tabId, button) {

            document.querySelectorAll('.question-container').forEach(el => el.classList.remove('active'));

            document.getElementById(tabId).classList.add('active');

            

            document.querySelectorAll('.nav-tab').forEach(el => el.classList.remove('active'));

            button.classList.add('active');

        }


        function checkAnswer(questionNum, correctAnswer) {

            const selectedOption = document.querySelector(`input[name="q${questionNum}"]:checked`);

            const resultMessage = document.getElementById('result-' + questionNum);

            const answerSection = document.getElementById('answer-' + questionNum);


            if (!selectedOption) {

                resultMessage.textContent = '選択肢を選んでください';

                resultMessage.className = 'result-message incorrect';

                resultMessage.style.display = 'block';

                answerSection.classList.remove('show');

                return;

            }


            const userAnswer = parseInt(selectedOption.value);

            

            if (userAnswer === correctAnswer) {

                resultMessage.textContent = '✓ 正解です！';

                resultMessage.className = 'result-message correct';

            } else {

                resultMessage.textContent = '✗ 不正解です。正答は第' + correctAnswer + '選択肢です。';

                resultMessage.className = 'result-message incorrect';

            }


            resultMessage.style.display = 'block';

            answerSection.classList.add('show');

        }
